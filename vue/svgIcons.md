#### SvgIcon.vue
```html
<template>
  <svg xmlns="http://www.w3.org/2000/svg" 
      v-bind:viewBox="viewBox" v-html="svgPaths">
    <slot></slot>
  </svg>
</template>


<script>
  import icons from "./iconList.js"

  export default {
    name: 'svg-icon',
    props: {
      icon: {
        type: String,
        required: false
      },
      viewBox: {
        type: String,
        required: false,
        default: "0 0 24 24",
      }
    },
    computed: {
      svgPaths: function () {    

        const iconList = icons;
        let symbol = "";        

        // 1 - if: check for icon prop 1st priority
        // 2 - try: check for slot value 2nd priority
        // 3 - catch: default fallback value ??

        if (this.icon) {
          symbol = this.icon;
        } else {          
          try {
            symbol = this.$slots.default[0].text;
          } catch {
            symbol = "earth"; 
          }          
        }

        const obj = iconList.find(item => item.name === symbol);
        
        return obj.path;
      }
    }
  }
</script>
```
